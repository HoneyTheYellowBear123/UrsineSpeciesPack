#yoinked from crusader spirit tee hee
#gives the scout trait to commanders and scientists
country_event = {
	id = honeybear_ursinespeciespack.110
	hide_window = yes

	is_triggered_only = yes

	immediate = {

		if = {
			limit = {
				has_valid_civic = civic_honeybear_ursinespeciespack_cub_scouts
			}
			every_owned_leader = {
				limit = {
					leader_class = { commander scientist }
				}
				give_trait_no_notify_ignore_autocracy_restriction = {
					TRAIT = leader_trait_scout
				}
			}
		}

	}
}

ship_event = {
	id = honeybear_ursinespeciespack_anomaly.2
	title = "honeybear_ursinespeciespack_anomaly.2.name"
	desc = "honeybear_ursinespeciespack_anomaly.2.desc"
	picture = GFX_evt_alien_nature
	show_sound = event_ship_bridge
	location = FROM

	is_triggered_only = yes

    #not wilderness
	option = {
		name = honeybear_ursinespeciespack_anomaly.2.a
		owner = {
			add_monthly_resource_mult = {
				resource = physics_research
				value = @tier3researchreward
				min = @tier3researchmin
				max = @tier3researchmax
			}
		}
	}

	#wilderness
	option = {
		name = honeybear_ursinespeciespack_anomaly.2.w
		trigger ={
		    owner = {
		        has_origin = origin_wilderness
		    }
		}
		owner = {
			add_monthly_resource_mult = {
				resource = physics_research
				value = @tier3researchreward
				min = @tier3researchmin
				max = @tier3researchmax
			}
		}
	}
}
